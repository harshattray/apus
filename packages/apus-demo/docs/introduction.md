## Quick Start

```tsx
import { BarChart } from 'apus';

const data = [
  { label: 'A', value: 10 },
  { label: 'B', value: 20 },
  { label: 'C', value: 15 }
];

function App() {
  return <BarChart data={data} />;
}
```
